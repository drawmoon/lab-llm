@contextmanager
def _acquire_session(host: str, port: int, user: str, password: str, space: str):
    from nebula3.gclient.net import ConnectionPool
    from nebula3.Config import Config
    s: Optional[Session] = None
    try:
        config = Config()
        config.max_connection_pool_size = 2
        connection_pool = ConnectionPool()
        assert connection_pool.init([(host, port)], config)
        s = connection_pool.get_session(user, password)
        s.execute("USE %s" % space)
        yield s
    finally:
        if s:
            s.release()
